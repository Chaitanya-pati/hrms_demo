    <script src="https://ajax.googleapis.com/ajax/libs/jquery/3.6.0/jquery.min.js"></script>
    <script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.0/dist/js/bootstrap.bundle.min.js"></script>

    <script>
        document.addEventListener('DOMContentLoaded', function () {
            var calendarEl = document.getElementById('calendar');
            var calendar;

            if (!calendarEl) {
                console.error("Calendar element not found.");
                return;
            }

            // Initialize Calendar
            calendar = new FullCalendar.Calendar(calendarEl, {
                initialView: 'dayGridMonth',
                selectable: true,
                selectMirror: true,
                events: fetchBookings,
                eventColor: '#FF6B6B',
                headerToolbar: {
                    left: 'prev,next today',
                    center: 'title',
                    right: 'dayGridMonth,timeGridWeek,timeGridDay'
                },
                dateClick: function (info) {
                    handleDateClick(info.dateStr);
                },
                eventClick: function(info) {
                    showEventDetails(info.event);
                },
                loading: function(bool) {
                    if (bool) {
                        showLoading();
                    } else {
                        hideLoading();
                    }
                }
            });

            calendar.render();

            // Event Listeners
            document.getElementById("test-button").addEventListener("click", syncAirbnb);
            document.getElementById("book").addEventListener("click", bookTest);
            document.getElementById("make-booking-button").addEventListener("click", makeBookingTest);

            // Utility Functions
            function showLoading() {
                $('#loadingSpinner').show();
            }

            function hideLoading() {
                $('#loadingSpinner').hide();
            }

            function showNotification(message, type = 'info', duration = 5000) {
                const alertClass = type === 'success' ? 'alert-success' : 
                                 type === 'error' ? 'alert-danger' : 'alert-info';
                const iconClass = type === 'success' ? 'fa-check-circle' : 
                                 type === 'error' ? 'fa-exclamation-circle' : 'fa-info-circle';
                
                const notification = $(`
                    <div class="alert ${alertClass} alert-dismissible fade show notification" role="alert">
                        <i class="fas ${iconClass} me-2"></i>
                        ${message}
                        <button type="button" class="btn-close" data-bs-dismiss="alert" aria-label="Close"></button>
                    </div>
                `);
                
                $('body').append(notification);
                
                if (duration > 0) {
                    setTimeout(() => {
                        notification.alert('close');
                    }, duration);
                }
            }

            // Handle Date Click
            function handleDateClick(dateStr) {
                showLoading();
                $.get('/Scheduler/AddBookingForm', { date: dateStr })
                    .done(function (html) {
                        $('#modalBody').html(html);
                        $('#bookingModal').modal('show');
                        
                        // Set the selected date in the form
                        const picker = window.currentDatePicker;
                        if (picker) {
                            picker.setDate(dateStr);
                        }
                    })
                    .fail(function(xhr, status, error) {
                        console.error('Error loading booking form:', error);
                        showNotification('Error loading booking form. Please try again.', 'error');
                    })
                    .always(function() {
                        hideLoading();
                    });
            }

            // Show Event Details
            function showEventDetails(event) {
                const eventInfo = `
                    <div class="event-details">
                        <h6>${event.title}</h6>
                        <p><strong>Start:</strong> ${event.start.toLocaleDateString()}</p>
                        <p><strong>End:</strong> ${event.end ? event.end.toLocaleDateString() : 'N/A'}</p>
                    </div>
                `;
                showNotification(eventInfo, 'info', 0);
            }

            // jQuery Ready Function
            $(document).ready(function () {
                // Handle Home dropdown change
                $(document).on('change', '#HomeId', function () {
                    const homeId = $(this).val();
                    if (!homeId) return;

                    // Hide the booking modal temporarily
                    $('#bookingModal').modal('hide');

                    // Load and show the Home Drawing modal
                    showLoading();
                    $.get(`/Scheduler/GetHomeDrawing?homeId=${homeId}`)
                        .done(function (html) {
                            $('#homeDrawingContainer').html(html);
                            
                            // Show the home drawing modal
                            const homeDrawingModal = new bootstrap.Modal(document.getElementById('homeDrawingModal'));
                            homeDrawingModal.show();

                            // When drawing modal closes, show booking modal again
                            $('#homeDrawingModal').on('hidden.bs.modal', function () {
                                setTimeout(() => {
                                    $('#bookingModal').modal('show');
                                }, 300);
                            });
                        })
                        .fail(function(xhr, status, error) {
                            console.error('Error loading home drawing:', error);
                            showNotification('Error loading room layout. Please try again.', 'error');
                            // Show booking modal again on error
                            $('#bookingModal').modal('show');
                        })
                        .always(function() {
                            hideLoading();
                        });
                });

                // Listen for successful booking saves
                $(window).on('bookingSaved', function(event, bookingData) {
                    calendar.refetchEvents();
                    showNotification('Booking saved successfully!', 'success');
                });

                // Listen for room selection updates
                $(window).on('roomsUpdated', function() {
                    showNotification('Room selection updated!', 'info', 3000);
                });
            });

            // API Functions
            function bookTest() {
                showLoading();
                $.ajax({
                    url: '/Scheduler/GetBookings',
                    type: 'GET',
                    success: function (data) {
                        console.log("Get Bookings Success:", data);
                        showNotification("Bookings retrieved successfully!", 'success');
                        calendar.refetchEvents();
                    },
                    error: function (xhr, status, error) {
                        console.error("Get Bookings Error:", error);
                        showNotification("Error retrieving bookings. Check console for details.", 'error');
                    },
                    complete: function() {
                        hideLoading();
                    }
                });
            }

            function makeBookingTest() {
                showLoading();
                $.ajax({
                    url: '/Scheduler/MakeBooking',
                    type: 'POST',
                    success: function (data) {
                        console.log("Make Booking Success:", data);
                        showNotification(data, 'success');
                        calendar.refetchEvents();
                    },
                    error: function (xhr, status, error) {
                        console.error("Make Booking Error:", error);
                        showNotification("Error making booking. Check console for details.", 'error');
                    },
                    complete: function() {
                        hideLoading();
                    }
                });
            }

            function fetchBookings(fetchInfo, successCallback, failureCallback) {
                $.ajax({
                    url: '/Scheduler/GetBookings',
                    type: 'GET',
                    dataType: 'json',
                    data: {
                        start: fetchInfo.startStr,
                        end: fetchInfo.endStr
                    },
                    success: function (data) {
                        let events = data.map(booking => ({
                            id: booking.id,
                            title: booking.title,
                            start: booking.start,
                            end: booking.end,
                            color: booking.color || '#FF6B6B',
                            display: booking.display || 'block',
                            extendedProps: {
                                customerName: booking.customerName,
                                customerEmail: booking.customerEmail,
                                homeId: booking.homeId
                            }
                        }));
                        successCallback(events);
                    },
                    error: function (xhr, status, error) {
                        console.error("❌ Error fetching bookings:", error);
                        showNotification("Error loading calendar events.", 'error');
                        failureCallback(error);
                    }
                });
            }

            function syncAirbnb() {
                showLoading();
                $.ajax({
                    url: '/Scheduler/SyncAirbnb',
                    type: 'GET',
                    dataType: 'json',
                    success: function (data) {
                        showNotification("✅ Airbnb Sync Successful! Calendar refreshed.", 'success');
                        calendar.refetchEvents();
                    },
                    error: function (xhr, status, error) {
                        console.error("❌ Airbnb sync failed:", error);
                        showNotification("❌ Failed to sync Airbnb. Check console for details.", 'error');
                    },
                    complete: function() {
                        hideLoading();
                    }
                });
            }
        });
    </script>
